

CMD Veya Terminal Kodları
Klasor
$ mkdir klasor ismi // Klasor Olusturma
$ ls //klasor içindekileri lister
$ ls -la // Gizli klasorleride listeler
$ dir //klasor içindekileri lister
$ git rm -r klasorAdı // Git Komutu İle Siliyoruz Yalnız .Git İcinde Olması Lazım
$ rm -rf klasor adı // Klasoru Siliyor
$ cd .. // Bir Önceki Klasor Yani Üst Klasor (parent)

Dosya
$ git mv DosyaAdı : yeniDosyaAdı // Dosya Adı Değişim Sağlar
$ mv DosyaAdı  yeniDosyaAdı // Git Olmadan Dosya Adı Değişim Sağlar 
		Bu Olayda Git Dosya Sildin Yeni Dosya Olusdurdun Gibi Algılıyor
$ git mv DosyaAdı YeniDosyaYolu/ // Dosya Tasıma İşlemi ilki Adı İkinci Yol Sonuna (/) Unutma
$ git mv DosyaAdı YeniDosyaYolu/YeniAd // Dosya Tası Ve Yeniden Adlandırma
$ touch dosyaadı.txt//  Yeni Dosya Olusturmak için Kullanılır
$ rm -rf deneme.txt // Dosya Silme
$ start dosyaAdı.txt // Dosya Acıyor
